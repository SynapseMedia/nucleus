21def has_valid_registered_service(): 

22 """ 

23 Check if pinata service is already registered 

24 :return: False if not registered else True 

25 """ 

26 registered_services = exec_command("/pin/remote/service/ls") 

27 registered_services_list = registered_services.get("RemoteServices") 

28 # Map resulting from registered services and search for "pinata" 

29 return any(map(lambda i: i["Service"] == PINATA_SERVICE, registered_services_list))

56def flush(limit=1000): 

57 """ 

58 Flush pinned entries from edge

60 """ 

61 pinned = pin_ls(limit) # Get current pin list from edge service 

62 logger.log.info(f"Flushing {pinned.get('count')} from edge") 

63 for _pin in pinned.get("results"): 

64 _pinned = _pin.get("pin")

66
