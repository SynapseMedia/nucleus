24class MediaScheme(DataObjectScheme): 

25 """ 

26 Generic abstract resource class definition 

27 :type route: Define how to reach the resource eg: cid | uri | path 

28 :type index: If subpath is needed index could be set as specific file path 

29 """ 

30 

31 route = fields.Str(required=True) # Could be cid | uri 

32 index = fields.Dict(required=False) # need refer to specific file path? 

33 

34 @validates("route") 

35 def validate_route(self, value): 

36 is_path = Path(value).exists() # Check for existing file path 

37 is_url = validators.url(value) # Check for valid url 

38 

39 if not is_url and not is_path and not is_cid(value): 

40 raise ValidationError("Route must be a CID | URI | Path")
